#Olvassuk be az airquality adatbázist. Vizsgáljuk az ózon mennyisége (Ozone) és a hőmérséklet (Temp) közötti kapcsolatot.

#Várhatóan mekkora lesz az ózonmennyiség 80 °F hőmérséklet esetén?

import = airquality
attach(import)
plot(Ozone ~ Temp)
model = lm(log(Ozone) ~ Temp)
summary(model)
# Mi a regressziós görbe egyenlete? 
# Ozone = exp(a + b*Temp)
a = model$coefficient[1]
b = model$coefficient[2]
f = function(x) exp(a + b*x)
